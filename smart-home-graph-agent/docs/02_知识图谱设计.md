│  └─────────┘    └─────────┘    └─────────┘    └─────────┘     │
│       │              │              │              │           │

│                         Step 5                                  │
│                      ┌─────────┐                                │

### 设计原则

```
┌─────────────────────────────────────────────────────────────────┐
│                    知识图谱设计原则                              │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│  1️⃣ 领域驱动 (Domain-Driven)                                   │
│     • 从业务需求出发，而非技术角度                               │
│     • 使用领域专家能理解的术语                                   │
│                                                                 │
│  2️⃣ 查询驱动 (Query-Driven)                                    │
│     • 先列出需要支持的查询                                       │
│     • 确保Schema能高效支持这些查询                               │
│                                                                 │
│  3️⃣ 适度抽象 (Right Level of Abstraction)                      │
│     • 不要过度泛化（如把所有东西都建模为"Thing"）                │
│     • 也不要过度具体（如为每个品牌建不同节点类型）                │
│                                                                 │
│  4️⃣ 关系优先 (Relationship-First)                              │
│     • 关系是图数据库的核心优势                                   │
│     • 有意义的关系应该显式建模                                   │
│                                                                 │
│  5️⃣ 可扩展性 (Extensibility)                                   │
│     • Schema应该能适应未来的变化                                 │
│     • 避免硬编码业务逻辑                                         │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

## 2.3 Cypher查询语言

### 基础语法

```cypher
-- ============================================
-- Cypher 基础语法速查
-- ============================================

-- 1. 节点模式
(n)                     -- 任意节点
(n:Label)              -- 带标签的节点
(n:Label {prop: val})  -- 带属性的节点

-- 2. 关系模式
-[r]->                 -- 有向关系
-[r:TYPE]->            -- 带类型的关系
-[r:TYPE {prop: val}]->-- 带属性的关系
<-[r]-                 -- 反向关系
-[r]-                  -- 无向关系

-- 3. 路径模式
(a)-[r]->(b)           -- 单跳
(a)-[*1..3]->(b)       -- 1-3跳
(a)-[:TYPE*]->(b)      -- 任意跳数
```

### CRUD操作

```cypher
-- ============================================
-- CREATE - 创建
-- ============================================

-- 创建节点
CREATE (r:Room {name: "客厅", floor: "一楼"})

-- 创建关系
MATCH (r:Room {name: "客厅"}), (d:Device {name: "智能电视"})
CREATE (r)-[:CONTAINS]->(d)

-- 创建带属性的关系
CREATE (scene)-[:USES_DEVICE {action: "调光至20%"}]->(device)


-- ============================================
-- READ - 查询
-- ============================================

-- 查询所有房间
MATCH (r:Room)
RETURN r.name, r.floor

-- 条件查询
MATCH (d:Device)
WHERE d.status = "online"
RETURN d.name, d.device_type

-- 关系查询
MATCH (r:Room)-[:CONTAINS]->(d:Device)
WHERE r.name = "客厅"
RETURN d.name

-- 多跳查询
MATCH (r:Room)-[:CONTAINS]->(d:Device)-[:HAS_CAPABILITY]->(c:Capability)
RETURN r.name, d.name, collect(c.name) AS capabilities


-- ============================================
-- UPDATE - 更新
-- ============================================

-- 更新属性
MATCH (d:Device {name: "智能电视"})
SET d.status = "offline"

-- 添加属性
MATCH (r:Room {name: "客厅"})
SET r.area = 30


-- ============================================
-- DELETE - 删除
-- ============================================

-- 删除节点（需先删除关系）
MATCH (d:Device {name: "旧设备"})
DETACH DELETE d

-- 删除关系
MATCH (r:Room)-[rel:CONTAINS]->(d:Device)
WHERE d.name = "旧设备"
DELETE rel
```

### 聚合与排序

```cypher
-- ============================================
-- 聚合函数
-- ============================================

-- 计数
MATCH (d:Device)
RETURN count(d) AS total_devices

-- 分组计数
MATCH (r:Room)-[:CONTAINS]->(d:Device)
RETURN r.name, count(d) AS device_count
ORDER BY device_count DESC

-- 收集为列表
MATCH (r:Room)-[:CONTAINS]->(d:Device)
RETURN r.name, collect(d.name) AS devices

-- 聚合多个字段
MATCH (r:Room)-[:CONTAINS]->(d:Device)-[:HAS_CAPABILITY]->(c:Capability)
RETURN r.name AS room,
       count(DISTINCT d) AS device_count,
       collect(DISTINCT c.name) AS all_capabilities
```

### 高级查询

```cypher
-- ============================================
-- 高级查询模式
-- ============================================

-- OPTIONAL MATCH (左外连接)
MATCH (r:Room)
OPTIONAL MATCH (r)-[:CONTAINS]->(d:Device)
RETURN r.name, count(d) AS device_count
-- 即使没有设备的房间也会返回

-- WITH 子句 (中间结果传递)
MATCH (r:Room)-[:CONTAINS]->(d:Device)
WITH r, count(d) AS cnt
WHERE cnt > 2
RETURN r.name, cnt

-- CASE 条件表达式
MATCH (d:Device)
RETURN d.name,
       CASE d.status
           WHEN "online" THEN "✅ 在线"
           WHEN "offline" THEN "❌ 离线"
           ELSE "❓ 未知"
       END AS status_text

-- 路径查询
MATCH path = (r1:Room)-[:ADJACENT_TO*1..3]->(r2:Room)
WHERE r1.name = "客厅"
RETURN r2.name, length(path) AS distance

-- 子查询 (CALL)
MATCH (r:Room)
CALL {
    WITH r
    MATCH (r)-[:CONTAINS]->(d:Device)
    RETURN count(d) AS cnt
}
RETURN r.name, cnt
```

## 2.5 实践查询示例

### 基础查询

```cypher
-- 查询1: 获取所有房间
MATCH (r:Room)
RETURN r.name AS 房间, r.floor AS 楼层, r.type AS 类型
ORDER BY r.floor, r.name

-- 查询2: 获取客厅的所有设备
MATCH (r:Room {name: "Living Room"})-[:CONTAINS]->(d:Device)
RETURN d.name AS 设备, d.device_type AS 类型, d.status AS 状态

-- 查询3: 获取所有能调光的设备
MATCH (d:Device)-[:HAS_CAPABILITY]->(c:Capability {name: "dim"})
RETURN d.name AS 设备, d.device_type AS 类型
```

### 多跳查询

```cypher
-- 查询4: 获取房间及其设备的能力 (2跳)
MATCH (r:Room)-[:CONTAINS]->(d:Device)-[:HAS_CAPABILITY]->(c:Capability)
WHERE r.name = "Living Room"
RETURN d.name AS 设备, collect(c.name) AS 能力列表

-- 查询5: 获取场景的完整配置 (多关系)
MATCH (s:Scene {name: "Movie Night"})-[:APPLIES_TO]->(r:Room)
OPTIONAL MATCH (s)-[u:USES_DEVICE]->(d:Device)
RETURN s.name AS 场景,
       r.name AS 房间,
       collect({设备: d.name, 操作: u.action}) AS 操作列表
```

### 复杂查询

```cypher
-- 查询6: 统计每个房间的设备数量
MATCH (r:Room)
OPTIONAL MATCH (r)-[:CONTAINS]->(d:Device)
RETURN r.name AS 房间,
       r.floor AS 楼层,
       count(d) AS 设备数量
ORDER BY count(d) DESC

-- 查询7: 找出能力最多的设备
MATCH (d:Device)-[:HAS_CAPABILITY]->(c:Capability)
WITH d, count(c) AS cap_count
ORDER BY cap_count DESC
LIMIT 5
RETURN d.name AS 设备, cap_count AS 能力数量

-- 查询8: 获取完整的房间上下文 (用于GraphRAG)
MATCH (r:Room {name: "Living Room"})-[:CONTAINS]->(d:Device)
OPTIONAL MATCH (d)-[:HAS_CAPABILITY]->(c:Capability)
WITH r, d, collect({name: c.name, description: c.description}) AS caps
WITH r, collect({
    device_id: d.device_id,
    name: d.name,
    type: d.device_type,
    capabilities: caps
}) AS devices
OPTIONAL MATCH (s:Scene)-[:APPLIES_TO]->(r)
RETURN {
    room: r.name,
    type: r.type,
    devices: devices,
    scenes: collect(DISTINCT s.name)
} AS context
```
